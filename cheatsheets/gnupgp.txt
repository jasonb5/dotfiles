Create new key, use `ECC and ECC` and 25519

gpg --expert --full-generate-key

E = encryption
S = signing
C = certification
A = authentication

List keys

gpg --list-keys

List secret keys

gpg --list-secret-keys

List longform

gpg --list-secret-keys --keyid-format=long

Add new subkey

gpg --expert --edit-key <key>

>> addkey

Export public key

gpg --armor --export <key>

Export keys

gpg --output <name>.public.gpg --export <key>
gpg --output <name>.secret.gpg --export-secret-key <key>
gpg --output <name>.secsub.gpg --export-secret-subkeys <key>
